import matplotlib.pyplot as plt

    terrain = Terrain("qBD4m7PNT5apV-Xl7PROxA")

    land = copy.deepcopy(elevation)

    ax.contourf(
        xs, ys, elevation, levels=[elevation.min(), 0], colors=["#212A2D"], zorder=1
    )

        zorder=4,
